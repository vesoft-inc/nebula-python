56class GraphType(BaseModel):
57 name: str
58 nodes: Dict[str, NodeType]
59 edges: Dict[str, EdgeType]
60
61 @classmethod
62 def from_models(
63 cls,
64 graph_type_name: str,
65 models: List[Union[Type["NodeModel"], Type["EdgeModel"]]],
66 ) -> Self:
67 from nebulagraph_python.orm.model import EdgeModel, NodeModel
68
69 nodes = {}
70 edges = {}
71 for model in models:
72 if issubclass(model, NodeModel):
73 nodes[model.get_type()] = model.to_type()
74 elif issubclass(model, EdgeModel):
75 edges[model.get_type()] = model.to_type()
76 return cls(name=graph_type_name, nodes=nodes, edges=edges)

78 def to_gql(self) -> str:
79 return (
80 f"CREATE GRAPH TYPE {self.name} {{\n"
81 + ",\n".join([x.to_gql() for x in self.nodes.values()])

83 + ",\n".join([x.to_gql() for x in self.edges.values()])
