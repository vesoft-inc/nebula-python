309 def cast(
310 self,
311 target_type: Optional[Type[T]] = None,
312 ) -> Union[T, TargetType]:
313 if self.value is None and target_type is None:
314 return None
315 target_type = ColumnToPy[self.data_type] if target_type is None else target_type # type: ignore
316
317 if target_type is not None:
318 if isinstance(self.value, target_type):
319 return self.value
320 else:
321 raise TypeError(
322 f"Cannot cast {type(self.value)} to {target_type}. ColumnType and value:",
323 self.data_type,
324 self.value,
325 )

327 def cast_primitive(self) -> Any:
328 """Convert the wrapped value to primitive Python types recursively.
329
330 For basic types, uses cast()
331 For composite types (Path, Node, Edge), calls cast_primitive()
332 For containers (Map, List, Record), recursively calls cast_primitive() on elements
333 """
334 outer = self.castcastcast()
335
336 # Handle composite types
337 if isinstance(outer, CompositeDataObject):
338 return outer.cast_primitive()
339
340 # Handle list recursively
341 if isinstance(outer, list):
342 ans = []
343 for i, v in enumerate(outer):
344 if isinstance(v, ValueWrapper):
345 ans.append(v.cast_primitive())
346 else:
347 raise TypeError(
348 f"Cannot cast list, where list[{i}] have type {type(v)}, not ValueWrapper. list[{i}]: ",
349 v,
350 )
351 return ans
352
353 return outer
