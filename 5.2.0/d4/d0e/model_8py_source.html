343def upsert_edge_only_gql(
344 src: Union[int, Dict[str, Any], NodeModel],
345 edge: EdgeModel,

347) -> str:

351 def _get_pattern(
352 idt: str, type_: str, element: Union[int, Dict[str, Any], NodeModel]
353 ) -> str:

367def _check_optional(t: Any) -> tuple[bool, type]:
368 """
369 Check if a type annotation is Optional[T], Union[T, None], or T | None.
370 Returns a tuple of (is_optional, inner_type).
371
372 Args:
373 t: A type annotation, typically from a pydantic model_field.annotation
374
375 Returns:
376 A tuple of (is_optional, inner_type) where:
377 - is_optional: True if the type is optional, False otherwise
378 - inner_type: The underlying type T (without the Optional wrapper)
379 """
380 # First, handle the case where t might be None itself
381 if t is None:
382 raise ValueError("t is None")
383
384 # Get origin and args - this works for both Python 3.8+ typing structures
385 origin = get_origin(t)
386 args = get_args(t)
387
388 # Handle both Union, type_extensions.Optional and the PEP 604 | operator
389 if origin is Union or origin is UnionType:
390 if len(args) == 2:
391 if (arg0 := args[0]) is type(None) or (arg1 := args[1]) is type(None):
392 return True, arg0 if arg0 is not type(None) else arg1
393 else:
394 raise ValueError("Union type should have exactly one None type")
395 else:
396 raise ValueError("Union type should have exactly two types")
397
398 # Handle standard types (not optional)
399 return False, t

479def _scan_metadata(t: type, field_info: FieldInfo) -> Optional[AdditionalMetadata]:

486def get_data_type(t: type, field_info: FieldInfo) -> str:
487 if t in DATA_TYPE_MAPPING:
488 return DATA_TYPE_MAPPING[t]
489 else:
490 # For list and List
491 if get_origin(t) is list:
492 inner_type = get_args(t)[0]
493 try:
494 inner_type_name = DATA_TYPE_MAPPING[inner_type]
495 except KeyError as e:
496 raise ValueError(
497 f"Unsupported inner type: {inner_type} for List"
498 ) from e
499 return f"LIST<{inner_type_name}>"
500
501 # For AsNVector and AsJSON
502 meta = _scan_metadata(t, field_info)
503 if meta is not None:
504 if meta.name == "AsNVector":
505 return f"VECTOR<{meta.data}, FLOAT>"
506 elif meta.name == "AsJSON":
507 return "STRING"
508 else:
509 raise NotImplementedError(f"Unsupported metadata: {meta.name}")
510 else:
511 raise ValueError(
512 f"No metadata found for the field {field_info}, which is not a primitive type."
513 )

547def _encode_properties(props: Mapping[str, FieldInfo]):
548 res: Dict[str, PropTypeRow] = {}
549 for k, v in props.items():
550 if k.startswith("nebula_obj__"):
551 continue
552 name = v.alias or k
553 is_optional, inner_type = _check_optional(v.annotation)
554 res[name] = PropTypeRow(
555 property_name=name,
556 data_type=get_data_type(inner_type, v),
557 nullable=is_optional,
558 default=None,
