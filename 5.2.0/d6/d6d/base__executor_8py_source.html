31def unwrap_value(v: Union[TargetType, TargetPrimitiveType]) -> str:
32 """Convert python value to GQL string"""
33
34 def from_vector(x: NVector):
35 return f"vector<{x.dimension}, float>([{', '.join(str(v) for v in x.values)}])"
36
37 mapping: Dict[
38 Type[Union[TargetType, TargetPrimitiveType]], Callable[[Any], str]
39 ] = {
40 str: lambda x: json.dumps(x, ensure_ascii=False).replace("'", "\\'"),
41 int: str,
42 float: str,
43 bool: lambda x: str(x).upper(),
44 list: lambda x: "[%s]" % ", ".join(unwrap_value(i) for i in x),
45 dict: lambda x: "{ %s }"
46 % ", ".join(f"`{k}`: {unwrap_value(v)}" for k, v in x.items()),
47 type(None): lambda x: "NULL",
48 NVector: from_vector,
49 date: lambda x: f'date("{x.strftime("%Y-%m-%d")}")',
50 }
51 try:
52 return mapping[type(v)](v)
53 except KeyError as e:
54 raise InternalError(f"Unsupported value type: {type(v)}") from e

57def unwrap_props(props: Mapping[str, Any]):
58 return ", ".join(f"`{k}`: {unwrap_value(v)}" for k, v in props.items())

126 def print_query_result(
127 self,
128 query: str,
129 style: str = "table",
130 width: Optional[int] = None,
131 min_width: int = 8,
132 max_width: Optional[int] = None,
133 padding: int = 1,
134 collapse_padding: bool = False,
135 ) -> None:
136 """Execute a query and print the results in a formatted way using rich
137
138 Args:
139 ----
140 query: The nGQL query to execute
141 style: Output style - either "table" (default) or "rows"
142 width: Fixed width for all columns. If None, width will be auto-calculated
143 min_width: Minimum width of columns when using table style
144 max_width: Maximum width of columns. If None, no maximum is enforced
145 padding: Number of spaces around cell contents in table style
146 collapse_padding: Reduce padding when cell contents are too wide
147
148 Raises:
149 ------
150 Exception if execution fails
151
152 """
153 try:
154 result = self.execute(query)
155 result.print(
156 style=style,
157 width=width,
158 min_width=min_width,
159 max_width=max_width,
160 padding=padding,
161 collapse_padding=collapse_padding,
162 )
163 except Exception as e:
164 from rich.console import Console
165 from rich.traceback import Traceback
166
167 console = Console()
168 console.print(f"[bold red]Error executing query:[/bold red] {e!s}")
169 if debug_flag:
170 console.print(Traceback.from_exception(type(e), e, e.__traceback__))
