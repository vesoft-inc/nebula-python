53 """
54 Create SSLParam instance for CA-signed certificates
55
56 Args:
57 ca_crt_file_path: Path to the CA certificate file
58 crt_file_path: Path to the certificate file
59 key_file_path: Path to the private key file
60
61 Returns:
62 SSLParam instance configured for CA-signed certificates
63 """

78 private_key = f.read()
79 else:
80 private_key = None
81
82 return cls(
83 ca_crt=ca_crt,
84 private_key=private_key,
85 cert=cert,
