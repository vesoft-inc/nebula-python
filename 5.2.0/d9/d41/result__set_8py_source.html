80 def _get_row_by_index(self, batch: Batch, index: int) -> Row:
81 """Parse row record from batch
82
83 Args:
84 ----
85 batch: the batch to parse from
86 index: the position of each vector in current batch
87
88 Returns:
89 -------
90 Row: row record
91
92 """
93
94 row = Row()
95 for i in range(batch.get_vectors_count()):
96 value = self.parserparser.decode_value_wrapper(

98 self.column_data_typescolumn_data_types[i],
99 index,
100 )

107 def rows(self):
108 """Generator that yields rows from the result table.
109
110 Returns
111 -------
112 Iterator[Row]: iterator of row records
113
114 Raises
115 ------
116 InternalError: if no result table data
117 """
118 for batch_index in range(self.num_batchesnum_batches):
119 current_batch = Batch(self.result_tableresult_table.batch[batch_index], self.byte_orderbyte_order)
120
121 # each VectorMetaData has the same numRecords value,
122 # just use the first one to get the numRecord for this batch
123 current_batch_row_size = 0
124 if current_batch.get_vectors_count() != 0:
125 current_batch_row_size = current_batch.get_batch_row_size()
126
127 # Skip empty batches
128 if current_batch.get_vectors_count() == 0:
129 continue
130
131 # Process rows in current batch
132 for row_index in range(current_batch_row_size):
133 row = self._get_row_by_index(current_batch, row_index)
134 yield row

214 def __init__(self, response: Any):
215 if not isinstance(response, ExecuteResponse):
216 raise InternalError(f"got {type(response)} object for server's response")

218 if response.HasField("result"):
219 self.result_tableresult_table = ResultTable(response.result)
220 self.column_namescolumn_names = self.result_tableresult_table.column_names
221 self.sizesize = self.result_tableresult_table.total_num_records
222 else:
223 self.result_tableresult_table = None
224 self.column_namescolumn_names = []
225 self.sizesize = 0
226
227 if not response.HasField("status"):
228 raise InternalError("status is not set in response")
229 self.status_codestatus_code = response.status.code.decode("utf-8")
230 self.status_messagestatus_message = response.status.message.decode("utf-8")
231 self.is_succeeded = self.status_codestatus_code == "00000"
232
233 if not response.HasField("summary"):
234 raise InternalError("summary is not set in response")
235 self.latency_uslatency_us = response.summary.elapsed_time.total_server_time_us
236 self.plan_descplan_desc = PlanInfoNode(response.summary.plan_info)
237 self.extra_infoextra_info = ExtraInfo(
238 cursor=response.cursor.decode("utf-8"),
239 affected_nodes=response.summary.query_stats.num_affected_nodes,
240 affected_edges=response.summary.query_stats.num_affected_edges,
241 total_server_time_us=response.summary.elapsed_time.total_server_time_us,
242 build_time_us=response.summary.elapsed_time.build_time_us,

244 serialize_time_us=response.summary.elapsed_time.serialize_time_us,

286 def as_pandas_df(self) -> "DataFrame":
287 """Convert result set to pandas DataFrame.
288
289 Returns
290 -------
291 pandas.DataFrame: DataFrame containing the query results
292
293 Raises
294 ------
295 ImportError: If pandas is not installed
296
297 """

301 raise ImportError(
302 "pandas is required to use this method. Please install it using 'pip install pandas'",
303 ) from e

306 raise RuntimeError(f"Query failed: {self.status_message}")
307
308 if self.sizesize == 0:
309 return pd.DataFrame(columns=self.column_namescolumn_names)

311 # Reset index to start
312 self._index = 0
313
314 # Build list of rows
315 rows = []
316 for record in self.records():

324 def as_ascii_table(
325 self,
326 console: Optional["Console"] = None,
327 style: str = "table",
328 width: Optional[int] = None,
329 min_width: int = 8,
330 max_width: Optional[int] = None,
331 padding: int = 1,
332 collapse_padding: bool = False,
333 ) -> Optional[str]:
334 """Print query results in a formatted table or row-by-row format. Return the string if console is not provided.
335
336 Args:
337 ----
338 console: rich.console.Console instance to use for printing. If None, a new instance will be created.
339 style: Output style - either "table" (default) or "rows"
340 width: Fixed width for all columns. If None, width will be auto-calculated
341 min_width: Minimum width of columns when using table style
342 max_width: Maximum width of columns. If None, no maximum is enforced
343 padding: Number of spaces around cell contents in table style
344 collapse_padding: Reduce padding when cell contents are too wide
345
346 Examples:
347 --------
348 # Print as table (default)
349 result.as_ascii_table()
350
351 # Print as rows
352 result.as_ascii_table(style="rows")
353
354 # Customize table formatting
355 result.as_ascii_table(width=20, max_width=30, padding=2)
356
357 Returns:
358 -------
359 Optional[str]: Formatted representation of the results, or error message if query failed.
360 If console is provided, the output will be printed to the console and None will be returned.
361
362 """
363 try:
364 from io import StringIO
365
366 from rich import box
367 from rich.console import Console
368 from rich.table import Table
369 except ImportError as e:
370 raise ImportError(
371 "The 'rich' library is required to use this method. Please install it using 'pip install rich'.",
372 ) from e
373
374 have_console: bool = console is not None
375 console = console or Console(file=StringIO(), force_terminal=False)
376
377 if not self.is_succeeded:
378 console.print(f"[bold red]Error:[/bold red] {self.status_message}")
379 if not have_console:
380 console.file.seek(0)
381 return console.file.read()
382
383 if self.sizesize == 0:
384 console.print("[yellow]Empty result set[/yellow]")
385 if not have_console:
386 console.file.seek(0)
387 return console.file.read()
388
389 # Reset index to start
390 if style == "rows":
391 # Row-by-row format
392 row_num = 1
393 for record in self.records():
394 console.print(f"\n[bold blue]Row {row_num}[/bold blue]")
395 for col, val in zip(self.column_namescolumn_names, record.values()):
396 console.print(f" [cyan]{col}:[/cyan] {val.cast_primitive()}")
397 row_num += 1
398
399 else:
400 # Table format
401 table = Table(
402 box=box.DOUBLE_EDGE,
403 show_header=True,
404 header_style="bold cyan",
405 width=width,
406 min_width=min_width,
407 padding=padding,
408 collapse_padding=collapse_padding,
409 highlight=True,
410 )
411
412 for header in self.column_namescolumn_names:
413 table.add_column(header, max_width=max_width, overflow="fold")
414
415 for record in self.records():
416 table.add_row(*[str(v.cast_primitive()) for v in record.values()])
417
418 console.print(table)
419
420 # Print summary
421 console.print("\n[bold green]Summary[/bold green]")
422 console.print(f" [green]Rows:[/green] {self.size}")
423 console.print(f" [blue]Latency:[/blue] {self.latency_us}s")
424
425 if not have_console:
426 console.file.seek(0)
427 return console.file.read()
