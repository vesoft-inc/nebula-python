48class NebulaAsyncClient(NebulaBaseAsyncExecutor):
49 """The async client for connecting to NebulaGraph. It is async/coroutine-level safe but not thread-safe,
50 which means you can not share the client instance across threads,
51 but you can call `await client.execute()` concurrently in async coroutines.
52
53 Required to explicitly call `close()` to release all resources.
54 """

65 @classmethod
66 async def connect(
67 cls,
68 hosts: Union[str, List[str], List[HostAddress]],
69 username: str,
70 password: Optional[str] = None,
71 *,
72 ssl_param: Union[SSLParam, Literal[True], None] = None,
73 auth_options: Optional[Dict[str, Any]] = None,
74 conn_config: Optional[ConnectionConfig] = None,
75 session_config: Optional[SessionConfig] = None,
76 session_pool_config: Optional[SessionPoolConfig] = None,
77 ):
78 """Connect to NebulaGraph and initialize the client
79
80 Args:
81 ----
82 hosts: Single host string ("hostname:port"), list of host strings,
83 or list of HostAddress objects
84 username: Username for authentication
85 password: Password for authentication
86 ssl_param: SSL configuration
87 auth_options: dict of authentication options
88 conn_config: Connection configuration. If provided, it overrides hosts and ssl_param.
89 session_config: Session configuration.
90 """
91 self = super().__new__(cls)
92 conn_conf = conn_config or ConnectionConfig.from_defults(hosts, ssl_param)
93 hosts = conn_conf.hosts
94 self._sessions_sessions = {}
95 if len(hosts) == 1:
96 self._conn_conn = AsyncConnection(conn_conf)
97 await self._conn_conn.connect()
98 else:
99 self._conn_conn = AsyncConnectionPool(conn_conf)
100 await self._conn_conn.connect()
101 try:
102 for host_addr in hosts:
103 conn = (
104 await self._conn_conn.get_connection(host_addr)
105 if isinstance(self._conn_conn, AsyncConnectionPool)
106 else self._conn_conn
107 )
108 if conn is None:
109 raise PoolError(
110 f"Failed to get connection to {host_addr} when initializing NebulaAsyncClient"
111 )
112 if session_pool_config:
113 self._sessions_sessions[host_addr] = await AsyncSessionPool.connect(
114 conn=conn,
115 username=username,
116 password=password,
117 auth_options=auth_options or {},
118 session_config=session_config or SessionConfig(),
119 pool_config=session_pool_config,
120 )
121 else:
122 self._sessions_sessions[host_addr] = AsyncSession(
123 conn=conn,
124 username=username,
125 password=password,
126 session_config=session_config or SessionConfig(),
127 auth_options=auth_options or {},
128 )
129 except Exception as e:
130 await self._conn_conn.close()
131 raise e
132 return self

134 async def execute(
135 self, statement: str, *, timeout: Optional[float] = None, do_ping: bool = False
136 ) -> ResultSet:
137 if isinstance(self._conn_conn, AsyncConnectionPool):
138 addr, _conn = await self._conn_conn.next_connection()
139 else:
140 addr = self._conn_conn.config.hosts[0]
141 _session = self._sessions_sessions[addr]
142
143 if isinstance(_session, AsyncSessionPool):
144 async with _session.borrow() as session:
145 return (
146 await session.execute(statement, timeout=timeout, do_ping=do_ping)
147 ).raise_on_error()
148 else:
149 return (
150 await _session.execute(statement, timeout=timeout, do_ping=do_ping)
151 ).raise_on_error()

153 async def ping(self, timeout: Optional[float] = None) -> bool:
154 try:
155 res = (
156 (await self.executeexecute(statement="RETURN 1", timeout=timeout))
157 .one()
158 .as_primitive()
159 )
160 if not res == {"1": 1}:

162 return True
163 except Exception:
