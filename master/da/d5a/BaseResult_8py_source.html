14class VertexData(object):

19 def __init__(self, row, col_names, decode_type='utf-8'):
20 """
21 the format is
22 '''
23 |tag_name._vid|tag_name.prop1|tag_name.prop2|
24 '''
25 """
26 if len(row.values) != len(col_names):
27 raise RuntimeError(
28 'Input row size is not equal with the col name size, {} != {}'.format(
29 row.values, col_names
30 )
31 )
32 self._row = row
33 self._decode_type = decode_type
34 self._col_names = []
35 self._tag_name = ''
36 for col_name in col_names:
37 names = col_name.split(b'.')
38 # TODO, just keep some behevior with before
39 if len(names) == 1 and names[0] == b'_vid':
40 continue
41 if len(names) != 2:
42 raise RuntimeError('Input wrong col name format of tag')
43 self._col_names.append(names[1])
44 self._tag_name = names[0]

46 def get_id(self):
47 """get vertex id, if the space vid_type is int, you can use get_id().as_int(),
48 if the space vid_type is fixed_string, you can use get_id().as_string()
49
50 :return: ValueWrapper
51 """
52 if len(self._row.values) < 1:
53 raise RuntimeError(
54 'The row value is bad format, '
55 'get vertex id failed: len is {}'.format(len(self._row.values))
56 )
57 return ValueWrapper(self._row.values[0], self._decode_type)

59 def as_node(self):
60 """convert the vertex data to structure Node
61
62 :return: Node
63 """
64 if len(self._row.values) < self.PROP_START_INDEX:
65 raise RuntimeError(
66 'The row value is bad format, '
67 'as node failed: len is {}'.format(len(self._row.values))
68 )
69
70 vertex = Vertex()
71 vertex.tags = []
72 vertex.vid = self._row.values[0]
73 tag = Tag()
74 tag.name = self._tag_name
75 tag.props = {}
76 index = self.PROP_START_INDEX
77 while index < len(self._col_names):
78 tag.props[self._col_names[index]] = self._row.values[index + 1]
79 index = index + 1
80 vertex.tags.append(tag)
81
82 return Node(vertex).set_decode_type(self._decode_type)
83
84 def get_prop_names(self):
85 """get all prop names from the vertex data
86
87 :return: list<string>
88 """
89 return self._col_names[self.PROP_START_INDEX :]
90
91 def get_prop_values(self):
92 """get all prop values from the vertex data
93
94 :return: list<ValueWrapper>
95 """
96 index = self.PROP_START_INDEX_with_vid
97 prop_values = []
98 while index < len(self._row.values):
99 prop_values.append(
100 ValueWrapper(self._row.values[index], decode_type=self._decode_type)
101 )
102 index = index + 1
103 return prop_values

232class BaseResult(object):
233 def __init__(self, data_sets: list, decode_type='utf-8', is_vertex=True):
234 assert data_sets is not None
235 self.is_vertex = is_vertex
236 self._data_sets = data_sets
237 self._decode_type = decode_type
238 self._pos = -1
239 self._data_set_pos = 0
240 self._table_pos = -1
241 self._size = 0
242 for data_set in self._data_sets:
243 self._size += len(data_set.rows)
244
245 def get_data_set(self):
246 """get DataSet, it's the origin values, the string type is binary
247
248 :return: DataSet
249 """
250 result = None
251 for data_set in self._data_sets:
252 if result is None:
253 result = data_set
254 continue
255 if len(data_set.column_names) != len(result.column_names):
256 raise RuntimeError('Multi DataSets are different col size')
257 result.rows.extend(data_set.rows)
258
259 return result
260
261 def get_data_set_wrapper(self):
262 """get DataSetWrapper, it's the wrapper for DataSet value, the string type is str
263
264 :return: DataSetWrapper
265 """
266 result = None
267 for data_set in self._data_sets:
268 if result is None:
269 result = data_set
270 continue
271 if len(data_set.column_names) != len(result.column_names):
272 raise RuntimeError('Multi DataSets are different col size')
273 result.rows.extend(data_set.rows)
274
275 if result is None:
276 return None
277 return DataSetWrapper(result, self._decode_type)
278
279 def __repr__(self):
280 return str(self._data_sets)
281
282 def __iter__(self):
283 self._pos = -1
284 return self

286 def __next__(self):
287 """The VertexData or EdgeData iterator
288
289 :return: VertexData or EdgeData Iterator
290 """
291 if len(self._data_sets) == 0 or self._pos >= self._size - 1:
292 raise StopIteration
293 self._pos += 1
294 self._table_pos += 1
295 if self._table_pos >= len(self._data_sets[self._data_set_pos].rows):
296 self._table_pos = 0
297 self._data_set_pos += 1
298 col_names = self._data_sets[self._data_set_pos].column_names
299 row = self._data_sets[self._data_set_pos].rows[self._table_pos]
300 if self.is_vertex:
301 return VertexData(row, col_names, self._decode_type)
302 else:
303 return EdgeData(row, col_names, self._decode_type)
